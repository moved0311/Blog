```py
def maximumTotalSum(self, maximumHeight: List[int]) -> int:
  res = []
  picked = set()

  for n in maximumHeight:
    tmp = n
    while tmp in picked:
      tmp -= 1
    if tmp <= 0:
      return -1
    picked.add(tmp)
    res.append(tmp)

  return sum(res)
```

```py
def minBitwiseArray(self, nums: List[int]) -> List[int]:
  ans = []
  N = len(nums)

  for i in range(N):
    for j in range(nums[i]):
      if j | j + 1 == nums[i]:
        ans.append(j)
        break
      if j == nums[i] - 1:
        ans.append(-1)
    return ans
```

```py
def minBitwiseArray(self, nums: List[int]) -> List[int]:
  for i, x in enumerate(nums):
    if x == 2:
      nums[i] = -1
    else:
      t = ~x
      lb = t & -t
      nums[i] ^= lb >> 1

    return nums
```
