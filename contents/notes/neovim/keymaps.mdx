## keymaps

`~/.config/nvim/lua/config/keymaps.lua`

```lua:keymaps.lua
local opts = { noremap = true, silent = true }
local keymap = vim.api.nvim_set_keymap

-- Insert
keymap("i", "jj", "<ESC>", opts)
```
